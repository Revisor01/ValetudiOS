@ObservationIgnored private var capabilitiesCache: [UUID: [String]] = [:]
@ObservationIgnored private var capabilitiesCacheDate: [UUID: Date] = [:]
private let capabilitiesTTL: TimeInterval = 86400 // 24 Stunden

2. Drei Methoden hinzufuegen:
```swift
func cachedCapabilities(for robotId: UUID) -> [String]? {
    guard let date = capabilitiesCacheDate[robotId],
          Date().timeIntervalSince(date) < capabilitiesTTL else { return nil }
    return capabilitiesCache[robotId]
}

func cacheCapabilities(_ capabilities: [String], for robotId: UUID) {
    capabilitiesCache[robotId] = capabilities
    capabilitiesCacheDate[robotId] = Date()
}

func invalidateCapabilities(for robotId: UUID) {
    capabilitiesCache.removeValue(forKey: robotId)
    capabilitiesCacheDate.removeValue(forKey: robotId)
}
```

2. **OTA-Reboot-Invalidierung:** Da RobotDetailViewModel bereits `updateService` als Property hat, fuege nach `setupUpdateService()` oder in `checkForUpdate()` eine Beobachtung hinzu. Nutze einen Task der den Phase-Wechsel beobachtet:

In der Methode wo `updateService` verwendet wird (z.B. `checkForUpdate()` ab Zeile 301, oder in einer neuen Hilfsmethode), fuege folgende Logik hinzu — am besten als separate Methode die einmalig aufgerufen wird:

```swift
/// Beobachtet UpdateService-Phase und invalidiert Capabilities-Cache nach OTA-Reboot
private func observeUpdatePhaseForCacheInvalidation() {
    // Merke letzte Phase um Transitionen zu erkennen
    var lastPhase = updateService?.phase ?? .idle
    
    // withObservationTracking im Task nutzen um Phase-Changes zu beobachten
    Task { [weak self] in
        while let self = self, let updateService = self.updateService {
            let currentPhase = updateService.phase
            // Transition von .rebooting zu .idle = erfolgreicher Reboot
            if lastPhase == .rebooting && currentPhase == .idle {
                self.robotManager.invalidateCapabilities(for: self.robot.id)
                logger.info("Capabilities cache invalidated after OTA reboot for robot \(self.robot.id)")
            }
            lastPhase = currentPhase
            
            // Warte auf naechste Phase-Aenderung via Observation
            try? await Task.sleep(for: .seconds(2))
            guard !Task.isCancelled else { return }
        }
    }
}
```
