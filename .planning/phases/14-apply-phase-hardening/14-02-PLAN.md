**1. Computed Property fuer Overlay-Sichtbarkeit:**
Am Anfang der RobotDetailView struct (bei den @State Properties), eine computed property hinzufuegen:

```swift
private var showUpdateOverlay: Bool {
    guard let phase = viewModel.updateService?.phase else { return false }
    switch phase {
    case .applying, .rebooting:
        return true
    default:
        return false
    }
}
```

**2. Overlay-View als private ViewBuilder:**
Innerhalb RobotDetailView, einen neuen `@ViewBuilder` hinzufuegen:
```swift
@ViewBuilder
private var updateOverlayView: some View {
    ZStack {
        Color.black.opacity(0.85)
            .ignoresSafeArea()
        
        VStack(spacing: 20) {
            ProgressView()
                .scaleEffect(1.5)
                .tint(.white)
            
            Text(updateOverlayTitle)
                .font(.title3)
                .fontWeight(.semibold)
                .foregroundStyle(.white)
            
            Text(updateOverlaySubtitle)
                .font(.subheadline)
                .foregroundStyle(.white.opacity(0.7))
                .multilineTextAlignment(.center)
                .padding(.horizontal, 40)
        }
    }
    .transition(.opacity)
}

private var updateOverlayTitle: String {
    if case .rebooting = viewModel.updateService?.phase {
        return String(localized: "update.rebooting_title")
    }
    return String(localized: "update.applying_title")
}

private var updateOverlaySubtitle: String {
    if case .rebooting = viewModel.updateService?.phase {
        return String(localized: "update.rebooting_hint")
    }
    return String(localized: "update.applying_hint")
}
```

**3. Overlay an den body anhaengen:**
Nach der `.alert(...)` Modifier-Chain, VOR `.task { await viewModel.loadData() }`:
```swift
.overlay {
    if showUpdateOverlay {
        updateOverlayView
    }
}
.navigationBarBackButtonHidden(showUpdateOverlay)
.interactiveDismissDisabled(showUpdateOverlay)
```

**4. Localization Keys:**
Folgende Keys werden in der View verwendet — sie muessen als `String(localized:)` Calls existieren. Die konkreten Texte:
- `update.applying_title` = "Update wird angewendet..."
- `update.applying_hint` = "Bitte die App nicht schliessen und das Geraet nicht ausschalten."
- `update.rebooting_title` = "Roboter startet neu..."
- `update.rebooting_hint` = "Bitte warten. Der Roboter meldet sich automatisch zurueck."

HINWEIS: Falls Localizable.xcstrings existiert, die Keys dort eintragen. Falls nicht, werden die `String(localized:)` Calls als Fallback die Key-Strings selbst anzeigen — das ist akzeptabel und wird in Phase 15 ggf. verbessert.

**5. Animation:**
Die `.overlay` mit `animation(.easeInOut, value: showUpdateOverlay)` wrappen:
```swift
.overlay {
    if showUpdateOverlay {
        updateOverlayView
    }
}
.animation(.easeInOut(duration: 0.3), value: showUpdateOverlay)
```
