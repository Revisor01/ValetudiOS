    **`src/app/page.tsx`** — Alle 18 Slides rendern:
    ```tsx
    import { screens } from "../data/screens"
    import { ScreenTemplate } from "../components/ScreenTemplate"
    import { toPng } from "html-to-image"

    export default function Page() {
      const handleExport = async (screenId: string, locale: string) => {
        const node = document.getElementById(`screen-${screenId}-${locale}`)
        if (!node) return
        const dataUrl = await toPng(node, { pixelRatio: 1, cacheBust: true })
        const link = document.createElement("a")
        link.download = `${screenId}-${locale}.png`
        link.href = dataUrl
        link.click()
      }

      return (
        <div>
          {/* Locale-Filter UI: DE | EN | Both */}
          {screens.map(s => (
            <div key={`${s.id}-${s.locale}`}>
              <button onClick={() => handleExport(s.id, s.locale)}>
                Export {s.id} ({s.locale.toUpperCase()})
              </button>
              <ScreenTemplate screen={s} />
            </div>
          ))}
        </div>
      )
    }
    ```

    **Export-Qualität:** `pixelRatio: 1` ist korrekt wenn Container bereits 1320×2868 (sonst 2640×5736 Output). `cacheBust: true` verhindert dass Mockup-PNG als veraltetes Asset gecacht wird.
